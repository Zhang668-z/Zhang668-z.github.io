    • @Override
    protected void actionExtendApplication(OaFiledispatchZz model, String workflowType) {
        // 补发进行的操作
        if (AppConstant.ACTION_ADD_SEND_UNIT.equals(workflowType)) {
            /* 将补发范围加到正式发文范围内*/
            // 获得发送范围
            String sendscope = model.getOafdSendscope();
            // 获得发送范围名字
            String sendscopename = model.getOafdSendscopename();

            sendscope = sendscope + "," + model.getStrMap().get("addScopeuuid").toString();
            sendscopename = sendscopename + "," + model.getStrMap().get("addScopename").toString();
            // 重新set发送范围
            model.setOafdSendscope(sendscope);
            model.setOafdSendscopename(sendscopename);
            // 进行数据库的更新
            this.getMyBatisDaoManager().update(model);

            // 记录补发流转信息
            WfHistory history = new WfHistory();
            // set项目案件UUID
            history.setWfhistoryItemuuid(model.getOafdUuid());
            // set当前处理用户   model.getOperator().getCractName()获得账户名称[登录用]  model.getOperator()获得账户对象
            history.setWfhistoryCurprocessor(model.getOperator().getCractName());
            // set后续处理业务
            history.setWfhistoryNextprocessor(model.getStrMap().get("addScopename").toString());
            // set当前执行动作
            history.setWfhistoryCurrentaction("补发");
            // set当前活动名称
            history.setWfhistoryCuractivityname("结束");

            this.getMyBatisDaoManager().save(history);

            /*对补发范围进行解析 外单位找到对应switchfile对象，调用它们的接口进行补发操作。*/
            // 交换中心文件(OaSwtichcenterFile)模型对象
            OaSwitchcenterFile switchFile = new OaSwitchcenterFile();
            // set源文件UUID
            switchFile.setOascfSourceuuid(model.getOafdUuid());
            OaSwitchcenterFile newSwitchfile = oaSwitchcenterFileService.getOaSwitchcenterFileBySourceuuid(switchFile);
            newSwitchfile.getStrMap().put("addScopeuuid", model.getStrMap().get("addScopeuuid").toString());
            newSwitchfile.getStrMap().put("addScopename", model.getStrMap().get("addScopename").toString());
            newSwitchfile.getStrMap().put("currentActionOs", "actionAddSend");
            newSwitchfile.getStrMap().put("subjectos", model.getOafdSubjectos());
            newSwitchfile.getStrMap().put("printno", model.getStrMap().get("printno"));
            newSwitchfile.getStrMap().put(AppConstant.STR_SMS, model.getStrMap().get("addSms"));
            newSwitchfile.getStrMap().put(AppConstant.STR_DD, model.getStrMap().get("addDd"));
            oaSwitchcenterFileService.doDocumentExecute(newSwitchfile);
        }else if (AppConstant.ACTION_FALSE_DELETE.equals(workflowType)) {
            // 工作流批量删除(假删除)
            actionFalseDeleteApplication(model);
        }
    }

    private void parseInnerOuterSendScope(OaFiledispatchZz model,WfEngine engine,String currentAction){
            //内部范围uuid
            String oafdScopeinner = model.getStrMap().get("oafdScopeinner");
            //外部范围uuid
            String oafdScopeouter = model.getStrMap().get("oafdScopeouter");
            StringBuilder sbUuid = new StringBuilder();
            StringBuilder sbName = new StringBuilder();
            String userUuids = "";
            String userNames = "";

            //内部范围
            if (StringUtils.hasText(oafdScopeinner)) {
                //根据混合节点，解析用户
                List<CoreAccount> coreAccounts = pubParseScopeService.parserScopeByMixUuidWithOrder(oafdScopeinner);
                for (CoreAccount coreAccount : coreAccounts){
                    sbUuid.append(coreAccount.getCractUuid()+",");
                    sbName.append(coreAccount.getCractName()+",");
                }
            }
            //外部范围
            if (StringUtils.hasText(oafdScopeouter)) {
                String[] uuids = oafdScopeouter.split(",");
                for (int i = 0; i < uuids.length; i++) {
                    String uuid = uuids[i];
                    CoreOrganization org = new CoreOrganization();
                    org.setCrorgUuid(uuid);
                    org = this.getMyBatisDaoManager().get(org,false);
                    if (org != null){
                        //单位
                        //根据机构的UUID获得机构的配置对象
                        Map m = pubOrgconfigService.getPubOrgconfig(org.getCrorgUuid());
                        if (!CollectionUtils.isEmpty(m)) {
                            // 进行不为null和""的判断
                            if (m.get("FileReceiverUuid") != null && !"".equals(m.get("FileReceiverUuid").toString())){
                                if (StringUtils.hasText(userUuids)){
                                    userUuids = userUuids + "," + m.get("FileReceiverUuid").toString();
                                }else {
                                    userUuids = m.get("FileReceiverUuid").toString();
                                }
                            }
                        }
                    }
                }
                // 进行切割
                String[] UuidList = userUuids.split(",");
                for (String receiverUuid : UuidList){
                    // 遍历
                    sbUuid.append(receiverUuid+",");
                    CoreAccount coreAccount = new CoreAccount();
                     // 通过uuid获取账户对象
                    coreAccount.setCractUuid(receiverUuid);
                    CoreAccount account = coreAccountService.get(coreAccount);
                    if (null != account) {
                        // 进行账户名称追加
                        sbName.append(account.getCractName()+ ",");
                    }
                }
            }
            if (sbUuid.length()>0) {
                // 根据索引删除指定的字符 ","
                userUuids = sbUuid.deleteCharAt(sbUuid.length() - 1).toString();
            }
            if (sbName.length()>0) {
                userNames = sbName.deleteCharAt(sbName.length() - 1).toString();
            }
            if (StringUtils.hasText(userUuids)){
                // 设置发送范围
                model.setOafdSendscope(userUuids);
            }
            if (StringUtils.hasText(userNames)){
                // 设置发送范围名字
                model.setOafdSendscopename(userNames);
            }
        }

工作流监控操作执行

  • @Override
    public void doWorkflowWatchExecute(OaFiledispatchZz model) {
        // 获得活动类型
        String workflowType = model.getStrMap().get("workflowType");
        String warning = "操作失败！";
        if (workflowType.equals("workflowBatchDelete")) {
            // 批量删除
            for (String uuid : model.getStrList()) {
                OaFiledispatchZz bean = new OaFiledispatchZz();
                bean.setUuid(uuid);// 设置UUID
                bean = this.getMyBatisDaoManager().exeSelectOneCommand(bean, "getFileBaseOne");
                if (bean != null) {
                    // 删除工作流信息
                    bean.getStrMap().put("workflowType", "workflowDelete");
                    // 执行工作流操作
                    wfEngineService.doWorkflowExecute(bean);
                    // 流程删除接口
                    workflowDeleteApplication(bean);
                }
            }
            warning = "处理完成！";
        } else if (workflowType.equals("workflowBatchFinish")) {
            for (String uuid : model.getStrList()) {
                OaFiledispatchZz bean = new OaFiledispatchZz();
                // 设置UUID
                bean.setUuid(uuid);
                bean = this.getMyBatisDaoManager().exeSelectOneCommand(bean, "getFileBaseOne");
                if (bean != null) {
                    // 结束工作流信息
                    bean.getStrMap().put("workflowType", "workflowRouteToTargetActivity");
                    bean.getStrMap().put("activityName", "结束");
                    bean.getStrMap().put("currentAction", "结束流程");
                    // 执行工作流操作
                    wfEngineService.doWorkflowExecute(bean);
                }
            }
            warning = "处理完成！";
            } else {
                warning = "找不到对应的处理方法！";
            }
        // 返回处理结果
        model.getStrMap().put("warning", warning);
    }
